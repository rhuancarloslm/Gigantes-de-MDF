52{
53 MOTOR_DDR |= (1 << INPUT1) | (1 << INPUT2) | (1 << INPUT4);
54 INPUT3_DDR |= (1 << INPUT3);
55 DDRC |= (1 << PC1);
56
57 // Configura LED's e laser como saída

59 LED_DDR |= (1 << LASER_PIN);
60 LED_PORT |= (1 << LED_PIN_1) | (1 << LED_PIN_2) | (1 << LED_PIN_3); // Inicialmente, todos LED's ligados
61
62 // Seta o receptor como entrada
63 LED_DDR &= ~(1 << RECEPTOR_PIN);
64 LED_PORT |= (1 << RECEPTOR_PIN); // Habilita pull-up
65
66 // Habilita a interrupção no receptor (INT0) - Reset dos LED's
67 EICRA |= (0b10 << ISC00); // Interrupção na borda de descida

73

75 TIMSK0 |= (1 << OCIE0A);
76 TCCR0A |= (1 << WGM01); // Modo CTC
